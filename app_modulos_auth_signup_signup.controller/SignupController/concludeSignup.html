@override
Future concludeSignup({@required String user, @required String pass}) async {
  try {
    dio.Response response = await repository.concludeSignup(user: user, pass: pass);
    await registerUser(
      UserModel(
        user: user,
        id: user,
        pass: pass,
        tokenJwt: SegAppConfig.instance.appController.auth.user.tokenJwt,
      ),
    );
  } catch (e) {
    await SegLoad().hide();
  }
}
